//DECLARATIONS
flagN, flagS : entier
cliN, cliS : chaine

//INIT
flagN <-- 0
flagS <-- 0
cliN, cliS <-- ""

//CORPS

OUVRIR NORD, SUD (LECTURE)

LIRE NORD
SI FF NORD Alors
	flagN <-- 1
	NORD.CLI <-- HIGH-VALUE

FSI

LIRE SUD
SI FF SUD Alors
	flagS <-- 1
	SUD.CLI <-- HIGH-VALUE
FSI

TANT QUE flagN = 0 ou flagS = 0

	SI NORD.CLI < SUD.CLI ALORS
		PERFORM TRT-NORD
		
	SINON SI  NORD.CLI > SUD.CLI ALORS
		PERFORM TRT-SUD
	    SINON
		PERFORM TRT-NORD
		PERFORM TRT-SUD
	   FINSI
	FINSI
FTQ


FIN


TRT-NORD.
	cliN <-- NORD.CLI
	TANT QUE cliN = NORD.Cli et flagN = 0
		AFF NORD.Mont

		LIRE NORD
		SI FF NORD Alors
			flagN <-- 1
			NORD.CLI <-- HIGH-VALUE
		FSI	
	FTQ

	.

TRT-SUD.
	cliS <-- SUD.CLI
	TANT QUE cliS = SUD.Cli et flagS = 0
		AFF SUD.Mont

		LIRE SUD
		SI FF SUD Alors
			flagS <-- 1
			SUD.CLI <-- HIGH-VALUE
		FSI	
	FTQ
	.
